00074     This macro finds the first (least significant) bit set in the
00075     input value.

00077     This macro is very similar to the function ffs() except that
00078     it evaluates its argument at compile-time, so it should only
00079     be applied to compile-time constant expressions where it will

00086     (least significant) bit set in the word val, or 0 if no bits are set.

00098      + (((x) & 017) == 0)       \
00099      + (((x) & 037) == 0)       \
00100      + (((x) & 077) == 0)       \
00101      + (((x) & 0177) == 0)      \
00102      + (((x) & 0377) == 0)      \
00103      + (((x) & 0777) == 0)      \
00104      + (((x) & 01777) == 0)     \
00105      + (((x) & 03777) == 0)     \
00106      + (((x) & 07777) == 0)     \
